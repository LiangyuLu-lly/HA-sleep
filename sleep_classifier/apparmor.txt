#include <tunables/global>

profile addon_local_sleep_classifier flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/python>

  # --- Init system (tini) + shell ---
  /sbin/tini rmix,
  /bin/bash rmix,
  /usr/bin/env rmix,
  /bin/sh rmix,

  # --- Runtime binaries ---
  # alpine 用 apk 装的 python 在 /usr/bin/，python:alpine base image 在 /usr/local/bin/。
  # 两条都 allow 以兼容两种 base image。
  /usr/local/bin/python3* rmix,
  /usr/bin/python3* rmix,
  /usr/bin/jq rmix,
  /usr/bin/find rmix,
  /bin/busybox rmix,

  # --- Application code ---
  /app/** r,
  /app/bootstrap_placeholders.py r,
  /app/render_effective_config.py r,
  /app/web_ui.py r,
  /app/requirements-runtime.txt r,
  /run.sh rix,

  # --- Persistent volumes (read-write) ---
  /data/** rwk,
  /share/** rwk,

  # --- Terminal / logging ---
  /dev/tty rw,
  /dev/null rw,
  /dev/urandom r,
  /dev/random r,

  # --- Network (Supervisor proxy 需要的最小集) ---
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  network unix stream,

  # --- Signals we need to receive ---
  signal (receive) set=(term, int, kill) peer=unconfined,
  signal (receive) set=(term, int) peer=@{profile_name},
  signal (send) set=(term, int, kill) peer=@{profile_name},

  # --- /proc 只读（logging / psutil 偶尔要看） ---
  /proc/*/status r,
  /proc/*/stat r,
  /proc/meminfo r,
  /proc/cpuinfo r,
  /proc/loadavg r,
  /proc/sys/kernel/random/uuid r,

  # --- tmp / pip cache fallback ---
  /tmp/ rw,
  /tmp/** rwk,

  # --- Timezone (tzdata) ---
  /etc/localtime r,
  /etc/timezone r,
  /usr/share/zoneinfo/** r,

  # --- Deny everything else by default (AppArmor 默认拒绝) ---
  # 无需显式 deny，profile 内未 allow 的路径/能力一律被拒。
}
